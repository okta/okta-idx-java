36  @Controller
37  public class LoginController {
38  
39      private final Logger logger = LoggerFactory.getLogger(LoginController.class);
40  
41      private static final String STATE = "state";
42      private static final String OTP = "otp";
43      private static final String NONCE = "nonce";
44      private static final String SCOPES = "scopes";
45      private static final String OKTA_BASE_URL = "oktaBaseUrl";
46      private static final String OKTA_CLIENT_ID = "oktaClientId";
47      private static final String REDIRECT_URI = "redirectUri";
48      private static final String ISSUER_URI = "issuerUri";
49      /* idx related properties */
50      private static final String INTERACTION_HANDLE = "interactionHandle";
51      private static final String CODE_VERIFIER = "codeVerifier";
52      private static final String CODE_CHALLENGE = "codeChallenge";
53      private static final String CODE_CHALLENGE_METHOD = "codeChallengeMethod";
54      private static final String CODE_CHALLENGE_METHOD_VALUE = "S256";
55      private static final String IDX_CLIENT_CONTEXT = "idxClientContext";
56  
57      private final OktaOAuth2Properties oktaOAuth2Properties;
58  
59      @Autowired
60      private IDXAuthenticationWrapper idxAuthenticationWrapper;
61  
62      private IDXClientContext idxClientContext;
63  
64      public LoginController(OktaOAuth2Properties oktaOAuth2Properties) {
65          this.oktaOAuth2Properties = oktaOAuth2Properties;
66      }
67  
68      @GetMapping(value = "/custom-login")
69      public ModelAndView handleLogin(HttpServletRequest request,
70                                      @RequestParam(name = "state", required = false) String state,
71                                      @RequestParam(name = "nonce") String nonce,
72                                      HttpSession session) throws MalformedURLException {
73  
74          logger.info("Handling login with state: {}, nonce {}", state, nonce);
75  
76          if (session.getAttribute(IDX_CLIENT_CONTEXT) == null) {
77              try {
78                  idxClientContext = idxAuthenticationWrapper.getClientContext();
79              } catch (ProcessingException e) {
80                  ModelAndView modelAndView = new ModelAndView("error");
81                  ErrorResponse errorResponse = e.getErrorResponse();
82                  if (errorResponse != null) {
83                      modelAndView.addObject("errorDetails",
84                              errorResponse.getError() + "," + errorResponse.getErrorDescription());
85                  } else {
86                      modelAndView.addObject("errorDetails", "Unknown error");
87                  }
88                  return modelAndView;
89              }
90              session.setAttribute(IDX_CLIENT_CONTEXT, idxClientContext);
91          }
92  
93          if (idxClientContext == null) {
94              ModelAndView modelAndView = new ModelAndView("error");
95              modelAndView.addObject("error_details", "Unknown error");
96              return modelAndView;
97          }
98  
99          // if we don't have the state parameter redirect
100         if (state == null) {
101             return new ModelAndView("redirect:" + oktaOAuth2Properties.getRedirectUri());
102         }
103 
104         String issuer = oktaOAuth2Properties.getIssuer();
105         // the widget needs the base url, just grab the root of the issuer
106         String orgUrl = new URL(new URL(issuer), "/").toString();
107 
108         ModelAndView mav = new ModelAndView("login");
109         mav.addObject(STATE, state);
110         mav.addObject(NONCE, nonce);
111         mav.addObject(SCOPES, oktaOAuth2Properties.getScopes());
112         mav.addObject(OKTA_BASE_URL, orgUrl);
113         mav.addObject(OKTA_CLIENT_ID, oktaOAuth2Properties.getClientId());
114         mav.addObject(INTERACTION_HANDLE, idxClientContext.getInteractionHandle());
115         mav.addObject(CODE_VERIFIER, idxClientContext.getCodeVerifier());
116         mav.addObject(CODE_CHALLENGE, idxClientContext.getCodeChallenge());
117         mav.addObject(CODE_CHALLENGE_METHOD, CODE_CHALLENGE_METHOD_VALUE);
118 
119         // from ClientRegistration.redirectUriTemplate, if the template is change you must update this
120         mav.addObject(REDIRECT_URI,
121                 request.getScheme() + "://" + request.getServerName() + ":" + request.getServerPort() +
122                         request.getContextPath() + "/authorization-code/callback"
123         );
124         mav.addObject(ISSUER_URI, issuer);
125 
126         session.setAttribute(CODE_VERIFIER, idxClientContext.getCodeVerifier());
127         return mav;
128     }
