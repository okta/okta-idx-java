112     /**
113      * Authenticate user with the supplied Authentication options (username and password) and
114      * returns the Authentication response object that contains:
115      * - IDX Client context
116      * - Token (access_token/id_token/refresh_token) object
117      * - Authentication status
118      * <p>
119      * Note: This requires 'Password' as the ONLY required factor in app Sign-on policy configuration.
120      *
121      * @param authenticationOptions the Authenticator options
122      * @param proceedContext the proceed context for the transaction
123      * @return the Authentication response
124      */
125     public AuthenticationResponse authenticate(AuthenticationOptions authenticationOptions, ProceedContext proceedContext) {
126         try {
127             // Check if identify flow needs to include credentials
128             boolean isIdentifyInOneStep = proceedContext.isIdentifyInOneStep();
129 
130             AuthenticationTransaction identifyTransaction = AuthenticationTransaction.proceed(client, proceedContext, () -> {
131                 IdentifyRequest identifyRequest;
132 
133                 if (isIdentifyInOneStep) {
134                     Credentials credentials = new Credentials();
135                     credentials.setPasscode(authenticationOptions.getPassword());
136 
137                     identifyRequest = IdentifyRequestBuilder.builder()
138                             .withIdentifier(authenticationOptions.getUsername())
139                             .withCredentials(credentials)
140                             .withStateHandle(proceedContext.getStateHandle())
141                             .build();
142                 } else {
143                     identifyRequest = IdentifyRequestBuilder.builder()
144                             .withIdentifier(authenticationOptions.getUsername())
145                             .withStateHandle(proceedContext.getStateHandle())
146                             .build();
147                 }
148 
149                 // identify user
150                 return client.identify(identifyRequest, proceedContext.getHref());
151             });
152 
153             AuthenticationResponse identifyResponse = identifyTransaction.asAuthenticationResponse();
154             if (isIdentifyInOneStep ||
155                     identifyResponse.getErrors() != null && !identifyResponse.getErrors().isEmpty()) {
156                 return identifyResponse;
157             }
158 
159             // If there are more than 1 authenticators, we need to allow users to choose one
160             if (identifyResponse.getAuthenticators() != null && identifyResponse.getAuthenticators().size() > 1) {
161                 return identifyResponse;
162             }
163 
164             AuthenticationTransaction passwordTransaction = selectPasswordOrEmailAuthenticatorIfNeeded(identifyTransaction);
165             if (Strings.isEmpty(authenticationOptions.getPassword())) {
166                 return passwordTransaction.asAuthenticationResponse(AuthenticationStatus.AWAITING_AUTHENTICATOR_VERIFICATION);
167             }
168             AuthenticationTransaction answerTransaction = passwordTransaction.proceed(() -> {
169                 // answer password authenticator challenge
170                 Credentials credentials = new Credentials();
171                 credentials.setPasscode(authenticationOptions.getPassword());
172 
173                 // build answer password authenticator challenge request
174                 AnswerChallengeRequest passwordAuthenticatorAnswerChallengeRequest =
175                         AnswerChallengeRequestBuilder.builder()
176                                 .withStateHandle(passwordTransaction.getStateHandle())
177                                 .withCredentials(credentials)
178                                 .build();
179 
180                 return passwordTransaction.getRemediationOption(RemediationType.CHALLENGE_AUTHENTICATOR)
181                         .proceed(client, passwordAuthenticatorAnswerChallengeRequest);
182             });
183             return answerTransaction.asAuthenticationResponse();
184         } catch (ProcessingException e) {
185             return handleProcessingException(e);
186         } catch (IllegalArgumentException e) {
187             return handleIllegalArgumentException(e);
188         }
189     }

191     /**
192      * Recover Password with the supplied username.
193      *
194      * @param username the username
195      * @param proceedContext the proceed context
196      * @return the Authentication response
197      */
198     public AuthenticationResponse recoverPassword(String username, ProceedContext proceedContext) {
199         try {
200             boolean isIdentifyInOneStep = proceedContext.isIdentifyInOneStep();
201 
202             if (isIdentifyInOneStep) {
203                 // recover
204                 AuthenticationTransaction recoverTransaction = AuthenticationTransaction.proceed(client, proceedContext, () -> {
205                     RecoverRequest recoverRequest = RecoverRequestBuilder.builder()
206                             .withStateHandle(getStateHandle(proceedContext.getIdxResponse().getCurrentAuthenticator().getValue().getRecover().getValue()))
207                             .build();
208 
209                     return client.recover(recoverRequest, null);
210                 });
211 
212                 RemediationOption remediationOption = recoverTransaction.getRemediationOption(RemediationType.IDENTIFY_RECOVERY);
213 
214                 IdentifyRequest identifyRequest = IdentifyRequestBuilder.builder()
215                         .withIdentifier(username)
216                         .withStateHandle(getStateHandle(remediationOption.form()))
217                         .build();
218 
219                 // identify user
220                 return recoverTransaction.proceed(() ->
221                         remediationOption.proceed(client, identifyRequest)
222                 ).asAuthenticationResponse(AuthenticationStatus.AWAITING_AUTHENTICATOR_SELECTION);
223             } else {
224                 // identify user
225                 AuthenticationTransaction identifyTransaction = AuthenticationTransaction.proceed(client, proceedContext, () -> {
226                     IdentifyRequest identifyRequest = IdentifyRequestBuilder.builder()
227                             .withIdentifier(username)
228                             .withStateHandle(proceedContext.getStateHandle())
229                             .build();
230                     return client.identify(identifyRequest, proceedContext.getHref());
231                 });
232                 IDXResponse identifyResponse = identifyTransaction.getResponse();
233 
234                 if (identifyResponse.getMessages() != null) {
235                     return identifyTransaction.asAuthenticationResponse(AuthenticationStatus.AWAITING_USER_EMAIL_ACTIVATION);
236                 }
237 
238                 // Check if instead of password, user is being prompted for list of authenticators to select
239                 if (identifyResponse.getCurrentAuthenticatorEnrollment() == null) {
240                     identifyTransaction = selectPasswordOrEmailAuthenticatorIfNeeded(identifyTransaction);
241                 }
242 
243                 Recover recover = identifyTransaction.getResponse()
244                         .getCurrentAuthenticatorEnrollment().getValue().getRecover();
245 
246                 AuthenticationTransaction finalIdentifyTransaction = identifyTransaction;
247                 AuthenticationTransaction recoverTransaction = identifyTransaction.proceed(() -> {
248                     // recover password
249                     RecoverRequest recoverRequest = RecoverRequestBuilder.builder()
250                             .withStateHandle(finalIdentifyTransaction.asAuthenticationResponse().getProceedContext().getStateHandle())
251                             .build();
252                     return recover.proceed(client, recoverRequest);
253                 });
254 
255                 return recoverTransaction.asAuthenticationResponse(AuthenticationStatus.AWAITING_AUTHENTICATOR_SELECTION);
256             }
257         } catch (ProcessingException e) {
258             return handleProcessingException(e);
259         } catch (IllegalArgumentException e) {
260             return handleIllegalArgumentException(e);
261         }
262     }

361     public AuthenticationResponse enrollAuthenticator(ProceedContext proceedContext, String authenticatorId) {
362         try {
363             AuthenticationResponse authenticationResponse =
364                     AuthenticationTransaction.proceed(client, proceedContext, () -> {
365                         Authenticator authenticator = new Authenticator();
366                         authenticator.setId(authenticatorId);
367 
368                         EnrollRequest enrollRequest = EnrollRequestBuilder.builder()
369                                 .withAuthenticator(authenticator)
370                                 .withStateHandle(proceedContext.getStateHandle())
371                                 .build();
372 
373                         return client.enroll(enrollRequest, proceedContext.getHref());
374                     }).asAuthenticationResponse();
375 
376             if (authenticationResponse.getWebAuthnParams() != null) {
377                 AuthenticatorEnrollments authenticatorEnrollments = authenticationResponse.getAuthenticatorEnrollments();
378 
379                 Optional<AuthenticatorEnrollment> authenticatorEnrollmentOptional = authenticatorEnrollments.stream()
380                         .filter(x -> "security_key".equals(x.getType()))
381                         .findAny();
382 
383                 authenticatorEnrollmentOptional.ifPresent(authenticatorEnrollment ->
384                         authenticationResponse.getWebAuthnParams().setWebauthnCredentialId(authenticatorEnrollment.getCredentialId()));
385             }
386 
387             return authenticationResponse;
388         } catch (ProcessingException e) {
389             return handleProcessingException(e);
390         } catch (IllegalArgumentException e) {
391             return handleIllegalArgumentException(e);
392         }
393     }

692     /**
693      * Populate UI form values for signing up a new user.
694      *
695      * @param proceedContext the proceedContext
696      * @return the authentication response
697      */
698     public AuthenticationResponse fetchSignUpFormValues(ProceedContext proceedContext) {
699         AuthenticationResponse newUserRegistrationResponse = new AuthenticationResponse();
700 
701         try {
702             // enroll new user
703             AuthenticationTransaction enrollTransaction = AuthenticationTransaction.proceed(client, proceedContext, () -> {
704                 EnrollRequest enrollRequest = EnrollRequestBuilder.builder()
705                         .withStateHandle(proceedContext.getStateHandle())
706                         .build();
707                 return client.enroll(enrollRequest, proceedContext.getSelectProfileEnrollHref());
708             });
709 
710             RemediationOption enrollProfileRemediationOption =
711                     enrollTransaction.getRemediationOption(RemediationType.ENROLL_PROFILE);
712 
713             List<FormValue> enrollProfileFormValues = enrollProfileRemediationOption.form().stream()
714                     .filter(x -> "userProfile".equals(x.getName()))
715                     .collect(Collectors.toList());
716 
717             List<FormValue> credentialFormValues = enrollProfileRemediationOption.form().stream()
718                     .filter(x -> "credentials".equals(x.getName()))
719                     .collect(Collectors.toList());
720 
721             List<FormValue> formValues = Stream.concat(enrollProfileFormValues.stream(), credentialFormValues.stream())
722                     .collect(Collectors.toList());
723 
724             newUserRegistrationResponse.setFormValues(formValues);
725             newUserRegistrationResponse.setProceedContext(enrollTransaction.createProceedContext());
726             return newUserRegistrationResponse;
727         } catch (ProcessingException e) {
728             return handleProcessingException(e);
729         } catch (IllegalArgumentException e) {
730             return handleIllegalArgumentException(e);
731         }
732     }

734     // If app sign-on policy is set to "any 1 factor", the next remediation after identify is
735     // select-authenticator-authenticate
736     // Check if that's the case, and proceed to select password authenticator
737     private AuthenticationTransaction selectPasswordOrEmailAuthenticatorIfNeeded(AuthenticationTransaction authenticationTransaction)
738             throws ProcessingException {
739         // If remediation contains challenge-authenticator for passcode, we don't need to check SELECT_AUTHENTICATOR_AUTHENTICATE
740         Optional<RemediationOption> challengeRemediationOptionOptional =
741                 authenticationTransaction.getOptionalRemediationOption(RemediationType.CHALLENGE_AUTHENTICATOR);
742 
743         if (challengeRemediationOptionOptional.isPresent()) {
744             // proceed with password challenge
745             return authenticationTransaction;
746         }
747 
748         Optional<RemediationOption> remediationOptionOptional =
749                 authenticationTransaction.getOptionalRemediationOption(RemediationType.SELECT_AUTHENTICATOR_AUTHENTICATE);
750         if (!remediationOptionOptional.isPresent()) {
751             // We don't need to.
752             return authenticationTransaction;
753         }
754         Map<String, String> authenticatorOptions = remediationOptionOptional.get().getAuthenticatorOptions();
755 
756         Authenticator authenticator = new Authenticator();
757         if (authenticatorOptions.get("password") != null) {
758             authenticator.setId(authenticatorOptions.get("password"));
759         }
760 
761         else if (authenticatorOptions.get("email") != null) {
762             authenticator.setId(authenticatorOptions.get("email"));
763         }
764 
765 
766         ChallengeRequest selectAuthenticatorRequest = ChallengeRequestBuilder.builder()
767                 .withStateHandle(authenticationTransaction.getStateHandle())
768                 .withAuthenticator(authenticator)
769                 .build();
770 
771         return authenticationTransaction.proceed(() ->
772                 remediationOptionOptional.get().proceed(client, selectAuthenticatorRequest)
773         );
774     }
