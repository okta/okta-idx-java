125     /**
126      * Get a key-value pair map of Authenticator options available for the current remediation option.
127      *
128      * where
129      * key - authenticator type (e.g. password, security_question, email)
130      * value - authenticator id (e.g. aut2ihzk2n15tsQnQ1d6)
131      *
132      * @return map of Authenticator type and id
133      */
134     public Map<String, String> getAuthenticatorOptions() {
135 
136         Map<String, String> authenticatorOptionsMap = new HashMap<>();
137 
138         FormValue[] formValues = this.form();
139 
140         Optional<FormValue> formValueOptional = Arrays.stream(formValues)
141             .filter(x -> "authenticator".equals(x.getName()))
142             .findFirst();
143 
144         if (formValueOptional.isPresent()) {
145             Options[] options = formValueOptional.get().options();
146 
147             for (Options option : options) {
148                 String key = null, val = null;
149                 FormValue[] optionFormValues = ((OptionsForm) option.getValue()).getForm().getValue();
150                 for (FormValue formValue : optionFormValues) {
151                     if (formValue.getName().equals("methodType")) {
152                         key = String.valueOf(formValue.getValue());
153                         StringBuilder nestedKeys = new StringBuilder();
154                         if (key.equals("null")) {
155                             // parse value from children
156                             for (Options children : formValue.options()) {
157                                 nestedKeys.append(children.getValue());
158                                 nestedKeys.append(",");
159                             }
160                             nestedKeys = nestedKeys.deleteCharAt(nestedKeys.length() - 1);
161                             key = nestedKeys.toString();
162                         }
163                     }
164                     if (formValue.getName().equals("id")) {
165                         val = String.valueOf(formValue.getValue());
166                     }
167                     if (formValue.getName().equals("enrollmentId")) {
168                         authenticatorOptionsMap.put("enrollmentId", String.valueOf(formValue.getValue()));
169                     }
170                 }
171                 if (key != null) {
172                     String[] tokenizedKey = key.split(",");
173                     if (tokenizedKey != null) {
174                         for (String tmpKey : tokenizedKey) {
175                             authenticatorOptionsMap.put(tmpKey, val);
176                         }
177                     } else {
178                         authenticatorOptionsMap.put(key, val);
179                     }
180                 }
181             }
182         }
183         return authenticatorOptionsMap;
184     }
