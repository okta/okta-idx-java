16  package com.okta.spring.example.controllers;
17  
18  import com.okta.idx.sdk.api.client.IDXAuthenticationWrapper;
19  import com.okta.idx.sdk.api.model.TokenType;
20  import com.okta.idx.sdk.api.response.TokenResponse;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.stereotype.Controller;
25  import org.springframework.web.bind.annotation.GetMapping;
26  
27  import javax.servlet.http.HttpSession;
28  
29  @Controller
30  public class LogoutController {
31  
32      /**
33       * logger instance.
34       */
35      private final Logger logger = LoggerFactory.getLogger(LogoutController.class);
36  
37      /**
38       * idx client instance.
39       */
40      @Autowired
41      private IDXAuthenticationWrapper idxAuthenticationWrapper;
42  
43      /**
44       * Handle logout by revoking the access token and invalidating the session.
45       *
46       * @param session the session
47       * @return the redirection to login view
48       */
49      @GetMapping("/logout")
50      public String logout(final HttpSession session) {
51          logger.info(":: Logout ::");
52  
53          // retrieve access token
54          TokenResponse tokenResponse =
55                  (TokenResponse) session.getAttribute("tokenResponse");
56  
57          if (tokenResponse != null) {
58              String accessToken = tokenResponse.getAccessToken();
59              // revoke access token
60              logger.info("Revoking access token");
61              idxAuthenticationWrapper.revokeToken(TokenType.ACCESS_TOKEN, accessToken);
62          }
63  
64          // invalidate session
65          session.invalidate();
66          return "redirect:/";
67      }
68  }
