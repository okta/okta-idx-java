103     /**
104      * Get a key-value pair map of Authenticator options available for the current remediation option.
105      *
106      * where
107      * key - authenticator type (e.g. password, security_question, email)
108      * value - authenticator id (e.g. aut2ihzk2n15tsQnQ1d6)
109      *
110      * @return map of Authenticator type and id
111      */
112     public Map<String, String> getAuthenticatorOptions() {
113 
114         Map<String, String> authenticatorOptionsMap = new HashMap<>();
115 
116         FormValue[] formValues = this.form();
117 
118         Optional<FormValue> formValueOptional = Arrays.stream(formValues)
119             .filter(x -> "authenticator".equals(x.getName()))
120             .findFirst();
121 
122         if (formValueOptional.isPresent()) {
123             Options[] options = formValueOptional.get().options();
124 
125             for (Options option : options) {
126                 String key = null, val = null;
127                 FormValue[] optionFormValues = ((OptionsForm) option.getValue()).getForm().getValue();
128                 for (FormValue formValue : optionFormValues) {
129                     if (formValue.getName().equals("methodType")) {
130                         key = String.valueOf(formValue.getValue());
131                         StringBuilder nestedKeys = new StringBuilder();
132                         if (key.equals("null")) {
133                             // parse value from children
134                             for (Options children : formValue.options()) {
135                                 nestedKeys.append(children.getValue());
136                                 nestedKeys.append(",");
137                             }
138                             nestedKeys = nestedKeys.deleteCharAt(nestedKeys.length() - 1);
139                             key = nestedKeys.toString();
140                         }
141                     }
142                     if (formValue.getName().equals("id")) {
143                         val = String.valueOf(formValue.getValue());
144                     }
145                     if (formValue.getName().equals("enrollmentId")) {
146                         authenticatorOptionsMap.put("enrollmentId", String.valueOf(formValue.getValue()));
147                     }
148                 }
149                 if (key != null) {
150                     authenticatorOptionsMap.put(key, val);
151                 }
152             }
153         }
154         return authenticatorOptionsMap;
155     }
