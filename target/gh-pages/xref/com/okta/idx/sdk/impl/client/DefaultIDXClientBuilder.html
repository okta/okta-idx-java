45  /**
46   * <p>The default {@link IDXClientBuilder} implementation. This looks for configuration files
47   * in the following locations and order of precedence (last one wins).</p>
48   * <ul>
49   * <li>classpath:com/okta/sdk/config/okta.properties</li>
50   * <li>classpath:com/okta/sdk/config/okta.yaml</li>
51   * <li>classpath:okta.properties</li>
52   * <li>classpath:okta.yaml</li>
53   * <li>~/.okta/okta.yaml</li>
54   * <li>Environment Variables (with dot notation converted to uppercase + underscores)</li>
55   * <li>System Properties</li>
56   * <li>Programmatically</li>
57   * </ul>
58   */
59  public class DefaultIDXClientBuilder implements IDXClientBuilder {
60  
61      private static final String ENVVARS_TOKEN   = "envvars";
62      private static final String SYSPROPS_TOKEN  = "sysprops";
63      private static final String OKTA_CONFIG_CP  = "com/okta/sdk/config/";
64      private static final String OKTA_YAML       = "okta.yaml";
65      private static final String OKTA_PROPERTIES = "okta.properties";
66  
67      private boolean allowNonHttpsForTesting = false;
68  
69      private ClientConfigurationuration.html#ClientConfiguration">ClientConfiguration clientConfig = new ClientConfiguration();
70  
71      public DefaultIDXClientBuilder() {
72          this(new DefaultResourceFactory());
73      }
74  
75      DefaultIDXClientBuilder(ResourceFactory resourceFactory) {
76          Collection<PropertiesSource> sources = new ArrayList<>();
77  
78          for (String location : configSources()) {
79  
80              if (ENVVARS_TOKEN.equalsIgnoreCase(location)) {
81                  sources.add(EnvironmentVariablesPropertiesSource.oktaFilteredPropertiesSource());
82              }
83              else if (SYSPROPS_TOKEN.equalsIgnoreCase(location)) {
84                  sources.add(SystemPropertiesSource.oktaFilteredPropertiesSource());
85              }
86              else {
87                  Resource resource = resourceFactory.createResource(location);
88  
89                  PropertiesSource wrappedSource;
90                  if (Strings.endsWithIgnoreCase(location, ".yaml")) {
91                      wrappedSource = new YAMLPropertiesSource(resource);
92                  } else {
93                      wrappedSource = new ResourcePropertiesSource(resource);
94                  }
95  
96                  PropertiesSource propertiesSource = new OptionalPropertiesSource(wrappedSource);
97                  sources.add(propertiesSource);
98              }
99          }
100 
101         Map<String, String> props = new LinkedHashMap<>();
102 
103         for (PropertiesSource source : sources) {
104             Map<String, String> srcProps = source.getProperties();
105             props.putAll(srcProps);
106         }
107 
108         if (Strings.hasText(props.get(DEFAULT_CLIENT_ISSUER_PROPERTY_NAME))) {
109             clientConfig.setIssuer(props.get(DEFAULT_CLIENT_ISSUER_PROPERTY_NAME));
110         }
111 
112         if (Strings.hasText(props.get(DEFAULT_CLIENT_ID_PROPERTY_NAME))) {
113             clientConfig.setClientId(props.get(DEFAULT_CLIENT_ID_PROPERTY_NAME));
114         }
115 
116         if (Strings.hasText(props.get(DEFAULT_CLIENT_SECRET_PROPERTY_NAME))) {
117             clientConfig.setClientSecret(props.get(DEFAULT_CLIENT_SECRET_PROPERTY_NAME));
118         }
119 
120         if (Strings.hasText(props.get(DEFAULT_CLIENT_SCOPES_PROPERTY_NAME))) {
121             Set<String> scopes = new HashSet<>(Arrays.asList(props.get(DEFAULT_CLIENT_SCOPES_PROPERTY_NAME).split("[\\s,]+")));
122             clientConfig.setScopes(scopes);
123         }
124 
125         if (Strings.hasText(props.get(DEFAULT_CLIENT_REDIRECT_URI_PROPERTY_NAME))) {
126             clientConfig.setRedirectUri(props.get(DEFAULT_CLIENT_REDIRECT_URI_PROPERTY_NAME));
127         }
128 
129         if (Strings.hasText(props.get(DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME))) {
130             allowNonHttpsForTesting = Boolean.parseBoolean(props.get(DEFAULT_CLIENT_TESTING_DISABLE_HTTPS_CHECK_PROPERTY_NAME));
131         }
132     }
133 
134     @Override
135     public IDXClientBuilder setIssuer(String issuer) {
136         this.clientConfig.setIssuer(issuer);
137         return this;
138     }
139 
140     @Override
141     public IDXClientBuilder setClientId(String clientId) {
142         this.clientConfig.setClientId(clientId);
143         return this;
144     }
145 
146     @Override
147     public IDXClientBuilder setClientSecret(String clientSecret) {
148         this.clientConfig.setClientSecret(clientSecret);
149         return this;
150     }
151 
152     @Override
153     public IDXClientBuilder setScopes(Set<String> scopes) {
154         this.clientConfig.setScopes(scopes);
155         return this;
156     }
157 
158     @Override
159     public IDXClientBuilder setRedirectUri(String redirectUri) {
160         this.clientConfig.setRedirectUri(redirectUri);
161         return this;
162     }
163 
164     @Override
165     public IDXClient build() {
166         this.validate();
167         return new BaseIDXClient(this.clientConfig, null);
168     }
169 
170     private void validate() throws IllegalArgumentException {
171         ConfigurationValidator.assertOrgUrl(clientConfig.getIssuer(), this.allowNonHttpsForTesting);
172         ConfigurationValidator.assertClientId(clientConfig.getClientId());
173         Assert.isTrue(!Collections.isEmpty(clientConfig.getScopes()), "At least one scope is required");
174         Assert.hasText(clientConfig.getRedirectUri(), "redirectUri is required");
175     }
176 
177     private static String[] configSources() {
178 
179         // lazy load the config sources as the user.home system prop could change for testing
180         return new String[] {
181             ClasspathResource.SCHEME_PREFIX + OKTA_CONFIG_CP + OKTA_PROPERTIES,
182             ClasspathResource.SCHEME_PREFIX + OKTA_CONFIG_CP + OKTA_YAML,
183             ClasspathResource.SCHEME_PREFIX + OKTA_PROPERTIES,
184             ClasspathResource.SCHEME_PREFIX + OKTA_YAML,
185             System.getProperty("user.home") + File.separatorChar + ".okta" + File.separatorChar + OKTA_YAML,
186             ENVVARS_TOKEN,
187             SYSPROPS_TOKEN
188         };
189     }
